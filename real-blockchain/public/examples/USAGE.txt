PharbitChain Integration Examples

React quickstart

import React from 'react';
import { PharbitClientProvider } from './PharbitClientProvider.jsx';
import WalletPanel from './WalletPanel.jsx';

export default function App(){
  return (
    <PharbitClientProvider baseUrl="http://localhost:3000/api">
      <WalletPanel />
    </PharbitClientProvider>
  );
}
